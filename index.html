function shareLineup() { const element = document.getElementById('share-section'); html2canvas(element, { windowHeight: element.scrollHeight, windowWidth: element.scrollWidth, scrollY: window.scrollY, scrollX: window.scrollX, foreignObjectRendering: true, imageTimeout: 2000 }).then(function(canvas) { const link = document.createElement('a'); link.href = canvas.toDataURL(); link.download = 'lineup.png'; link.click(); }); }
